do you have added the --debug clause in the init script?

/etc/init.d/rdiff-web
#--------------------
       echo -n "Starting rdiffWeb..."
       /usr/bin/rdiff-web --debug --pid-file="$PIDFILE" \
       --background > /dev/null 2>&1
#--------------------
